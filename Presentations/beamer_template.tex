%% ============================================================
%% TITLE PAGE
%% Beamer's text area is narrower than \paperwidth by ~2x28.5pt.
%% We cancel its left inset with a negative \hspace, then work in
%% a \paperwidth-wide minipage so rules and content align correctly.
%% Frame must be [t, plain, noframenumbering].
%% ============================================================
\setbeamertemplate{title page}{%
    % Cancel Beamer's left margin (= (paperwidth - textwidth) / 2 = ~28.45pt)
    % so our paperwidth-wide minipage starts at the true paper left edge.
    \hspace*{-28.45pt}%
    \begin{minipage}[t][\paperheight][t]{\paperwidth}%
        \vspace{1.2cm}%
        % ---- Title ----
        \noindent\hspace{1cm}{\Large\textbf{\inserttitle}}\par%
        \vspace{0.3cm}%
        % ---- Subtitle ----
        \noindent\hspace{1cm}{\large\insertsubtitle}\par%
        \vspace{0.55cm}%
        % ---- Red line  (1cm margin both sides) ----
        \noindent\hspace{1cm}\textcolor{PrimaryRed}{\rule{\dimexpr\paperwidth-2cm\relax}{2pt}}\\%
        \vspace{0.01cm}%
        % ---- Blue line ----
        \noindent\hspace{1cm}\textcolor{PrimaryBlue}{\rule{\dimexpr\paperwidth-2cm\relax}{2pt}}\par%
        \vspace{0.4cm}%
        % ---- Author (left) + logos (right) ----
        \noindent\hspace{1cm}%
        \begin{minipage}[t]{0.58\paperwidth}%
            {\normalsize\insertauthor}\par%
            \vspace{0.1cm}%
            {\small\insertinstitute}\par%
            \vspace{0.1cm}%
            {\small\insertdate}%
        \end{minipage}%
        \begin{minipage}[t]{\dimexpr0.42\paperwidth-2cm\relax}%
            \raggedleft
            % Replace \fbox with \includegraphics when logos are ready:
            % \includegraphics[height=1.1cm]{logo1.png}\hspace{0.3cm}%
            % \includegraphics[height=1.1cm]{logo2.png}%
            \fbox{\parbox[c][1.1cm][c]{3cm}{\centering\tiny logo space}}%
        \end{minipage}\par%
    \end{minipage}%
}
