%% ============================================================
%% FOOTER  —  red line / white gap / blue line + author | date | n/N
%% ============================================================
\setbeamertemplate{footline}{%
    \vskip-0.6cm%
    \hbox{\hspace{1cm}\textcolor{RuleColorA} {\rule{\dimexpr\paperwidth-2cm\relax}{1.5pt}}}%
    \vskip3pt%
    \hbox{\hspace{1cm}\textcolor{RuleColorB}{\rule{\dimexpr\paperwidth-2cm\relax}{1.5pt}}}%
    \vskip0.08cm%
    \leavevmode%
    \hbox{%
        \begin{beamercolorbox}[wd=0.33\paperwidth,ht=0.3cm,dp=0.1cm,left,leftskip=1cm]{footline}
            \usebeamerfont{footline}\insertshortauthor
        \end{beamercolorbox}%
        \begin{beamercolorbox}[wd=0.34\paperwidth,ht=0.3cm,dp=0.1cm,center]{footline}
            \usebeamerfont{footline}\insertshortdate
        \end{beamercolorbox}%
        \begin{beamercolorbox}[wd=0.33\paperwidth,ht=0.3cm,dp=0.1cm,right,rightskip=1cm]{footline}
            \usebeamerfont{footline}\insertframenumber{} / \inserttotalframenumber
        \end{beamercolorbox}%
    }%
    \vskip2pt%
}

%% ============================================================
%% TITLE PAGE
%% Beamer's text area is narrower than \paperwidth by ~2x28.5pt.
%% We cancel its left inset with a negative \hspace, then work in
%% a \paperwidth-wide minipage so rules and content align correctly.
%% Frame must be [t, plain, noframenumbering].
%% ============================================================
\setbeamertemplate{title page}{%
    % Cancel Beamer's left margin (= (paperwidth - textwidth) / 2 = ~28.45pt)
    % so our paperwidth-wide minipage starts at the true paper left edge.
    \hspace*{-28.45pt}%
    \begin{minipage}[t][\paperheight][t]{\paperwidth}%
        \vspace{1.2cm}%
        % ---- Title ----
        \noindent\hspace{1cm}{\Large\textbf{\inserttitle}}\par%
        \vspace{0.3cm}%
        % ---- Subtitle ----
        \noindent\hspace{1cm}{\large\insertsubtitle}\par%
        \vspace{0.55cm}%
        % ---- Red + Blue lines (1cm margin both sides) ----
        \hbox{\hspace{1cm}\textcolor{RuleColorA}{\rule{\dimexpr\paperwidth-2cm\relax}{2pt}}}%
        \nointerlineskip\vskip2pt%
        \hbox{\hspace{1cm}\textcolor{RuleColorB}{\rule{\dimexpr\paperwidth-2cm\relax}{2pt}}}%
        \vspace{0.4cm}%
        % ---- Author (left) + logos (right) ----
        \noindent\hspace{1cm}%
        \begin{minipage}[c]{0.58\paperwidth}%
            {\small\textcolor{DarkGrey}{\insertauthor}}\par%
            \vspace{0.1cm}%
            {\footnotesize\textcolor{DarkGrey}{\insertinstitute}}\par%
            \vspace{0.1cm}%
            {\footnotesize\textcolor{DarkGrey}{\insertdate}}%
        \end{minipage}%
        \begin{minipage}[c]{\dimexpr0.42\paperwidth-2cm\relax}%
            \raggedleft
            \includegraphics[height=1.8cm]{logo.png}%
        \end{minipage}\par%
    \end{minipage}%
}
